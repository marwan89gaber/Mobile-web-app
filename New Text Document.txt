    <!-- Troll Popup Script -->
    <script>
        let trollPopupShown = false;

        function openTrollPopup(event) {
            if (trollPopupShown) return;
            trollPopupShown = true;

            // Prevent default navigation
            event.preventDefault();
            
            const popup = document.getElementById("trollPopup");
            const video = document.getElementById("trollVideo");
            const unmuteBtn = document.getElementById("trollUnmuteBtn");
            
            // Show popup
            popup.classList.add("troll-popup-open");
            
            // Play video (muted initially - required by browsers)
            setTimeout(function() {
                video.play().catch(err => console.log("Play error:", err));
            }, 300);
            
            // Unmute button handler
            unmuteBtn.addEventListener('click', function() {
                video.muted = false;
                video.volume = 1.0;
                unmuteBtn.style.display = 'none';
                if (video.paused) {
                    video.play();
                }
            });
            
            // Also allow clicking video to unmute
            video.addEventListener('click', function() {
                if (video.muted) {
                    video.muted = false;
                    video.volume = 1.0;
                    unmuteBtn.style.display = 'none';
                }
            });
        }

        function closeTrollPopup() {
            const popup = document.getElementById("trollPopup");
            const video = document.getElementById("trollVideo");
            popup.classList.remove("troll-popup-open");
            video.pause();
            video.currentTime = 0;
            trollPopupShown = false;
        }

        function showSecondPopup() {
            // Close first popup
            const firstPopup = document.getElementById("trollPopup");
            const firstVideo = document.getElementById("trollVideo");
            firstPopup.classList.remove("troll-popup-open");
            firstVideo.pause();
            firstVideo.currentTime = 0;
            
            // Show second popup
            const secondPopup = document.getElementById("secondTrollPopup");
            const secondVideo = document.getElementById("secondTrollVideo");
            const secondUnmuteBtn = document.getElementById("secondTrollUnmuteBtn");
            
            secondPopup.classList.add("troll-popup-open");
            
            // Play video
            setTimeout(function() {
                secondVideo.play().catch(err => console.log("Play error:", err));
            }, 300);
            
            // Unmute button handler
            secondUnmuteBtn.addEventListener('click', function() {
                secondVideo.muted = false;
                secondVideo.volume = 1.0;
                secondUnmuteBtn.style.display = 'none';
                if (secondVideo.paused) {
                    secondVideo.play();
                }
            });
            
            // Also allow clicking video to unmute
            secondVideo.addEventListener('click', function() {
                if (secondVideo.muted) {
                    secondVideo.muted = false;
                    secondVideo.volume = 1.0;
                    secondUnmuteBtn.style.display = 'none';
                }
            });
            
            // Countdown and redirect
            let countdown = 5;
            const countdownElement = document.getElementById("countdown");
            
            const countdownInterval = setInterval(function() {
                countdown--;
                countdownElement.textContent = countdown;
                
                if (countdown <= 0) {
                    clearInterval(countdownInterval);
                    window.location.href = "./val.html";
                }
            }, 1000);
        }

        // Intercept all clicks on links and buttons
        document.addEventListener('DOMContentLoaded', function() {
            // Get all links and buttons
            const allLinks = document.querySelectorAll('a[href]');
            const allButtons = document.querySelectorAll('button[type="submit"], button:not(.troll-btn):not(.search__close-btn):not(.close-btn):not(#trollUnmuteBtn)');
            
            // Add click interceptor to all links
            allLinks.forEach(function(link) {
                link.addEventListener('click', function(e) {
                    // Only intercept if not already shown
                    if (!trollPopupShown) {
                        openTrollPopup(e);
                    }
                });
            });
            
            // Add click interceptor to all buttons
            allButtons.forEach(function(button) {
                button.addEventListener('click', function(e) {
                    // Only intercept if not already shown
                    if (!trollPopupShown) {
                        openTrollPopup(e);
                    }
                });
            });
        });
    </script>
